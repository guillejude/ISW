    public VentanaVerSorteos(VentanaPrincipal ventanaPrincipal, Sistema sistema) {
        initComponents();
        this.padre = ventanaPrincipal;
        this.sistema = sistema;
        DefaultTableModel table = (DefaultTableModel) tblSorteos.getModel();
        DateFormat df = new SimpleDateFormat("dd-MM-yyyy");
        table.setRowCount(this.sistema.getCantSorteos());
        int k = 0;
        for(int i = 0; i<this.sistema.getRestaurantes().size(); i++){
            Restaurante restaurante = this.sistema.getRestaurantes().get(i);
            for(int j = 0; j<restaurante.getSorteos().size(); j++){
                Sorteo elSorteo = (Sorteo) restaurante.getSorteos().get(j);
                tblSorteos.setValueAt(elSorteo.getNombre(), k, 0);
                tblSorteos.setValueAt(df.format(elSorteo.getFechaInicio()), k, 1);
                tblSorteos.setValueAt(df.format(elSorteo.getFechaInicio()), k, 2);
                tblSorteos.setValueAt(elSorteo.getCantidadGanadores(), k, 3);
                String realizado = "NO";
                if(elSorteo.isRealizado()){
                    realizado = "SI";
                }
                tblSorteos.setValueAt(realizado, k, 4);
                k++;
            }
        }
        setDefaultCloseOperation(DO_NOTHING_ON_CLOSE);
    }
